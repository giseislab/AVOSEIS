Installation procedure for the AVOSEIS package on your filesystem:

# If $AVOSEIS and $AVOSEISMAKE already exist:
	make update

# Otherwise:
	# Create the directory where AVOSEIS will be installed, e.g. /usr/local/avoseis/5.0-64
	sudo mkdir -p /usr/local/avoseis/5.0-64
	If using bash:
		export AVOSEIS=/usr/local/avoseis/5.0-64
	If using csh
		setenv AVOSEIS "/usr/local/avoseis/5.0-64"
	Build AVOSEIS:
		cd adm/coldstart
		make
		cd ../..
		mkdir -p ${AVOSEIS}/man/man1
		mkdir -p ${AVOSEIS}/man/man3p
		mkdir -p ${AVOSEIS}/man/man5
		mkdir -p ${AVOSEIS}/data/perl/Avoseis
		make update

# Additional steps for real-time servers only
	make host 
	make html

# For a really clean build, do this:
	chmod -R a+w $AVOSEIS
	cd lib/makerules
	make Include
	cd ../..
	make update
	chmod -R a-w $AVOSEIS
